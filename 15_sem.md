# Диаграмма работы склада 

```mermaid
---
title: Работа склада (EHSM)
config:
    theme: neo
    look: neo
---
stateDiagram-v2
    direction TB

    state maintain <<fork>>
    [*] --> maintain
    maintain --> Склад
    maintain --> Логистика

    note left of maintain
        Начальный остаток: 1000
        Порог заказа: < 600
        Доставка: 5 раб. дней
    end note

    state Склад {
        direction TB
        [*] --> ПН
        
        %% Понедельник: Если была доставка, здесь происходит учет товара
        ПН : Понедельник<br>exit/ спрос=Rand(40,63)<br>остаток-=спрос <br>entry/ если_доставка_была_то_учет
        ВТ : Вторник<br>exit/ спрос=Rand(40,63)<br>остаток-=спрос
        СР : Среда<br>exit/ спрос=Rand(40,63)<br>остаток-=спрос
        ЧТ : Четверг<br>exit/спрос=Rand(40,63)<br>остаток-=спрос
        ПТ : Пятница<br>exit/ спрос=Rand(40,63)<br>остаток-=спрос <br>Инвентаризация
        
        ПН --> ВТ
        ВТ --> СР
        СР --> ЧТ
        ЧТ --> ПТ
        
        state ПроверкаОстатка <<choice>>
        
        ПТ --> ПроверкаОстатка : Инвентаризация
        
        ПроверкаОстатка --> Логистика.Ожидание : Остаток < 600
        
        ПроверкаОстатка --> ПН : Остаток >= 600
    }

    state Логистика {
        direction TB
        [*] --> Ожидание
        
        Ожидание --> ЗаказАктивен : Старт_заказа
        
        state ЗаказАктивен {
            direction LR
            [*] --> Расчет
            
            Расчет : entry/ <BR>Объем_Заказа=1000-Остаток
            Расчет --> Доставка_5_дней
            
            Доставка_5_дней : <u>Доставка</u><br>(5 рабочих дней)
            
            Доставка_5_дней --> [*]
        }
        
        ЗаказАктивен --> Склад.ПН : Доставка_завершена_переход_в_ПН
    }

    note right of Склад.ПН
        В этот понедельник поступает товар,
        заказанный 5 дней назад.
    end note
